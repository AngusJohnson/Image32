procedure DrawGlow(img: TImage32; const polygons: TPathsD; fillRule: TFillRule; color: TColor32; blurRadius: integer); overload;


  uses Img32, Img32.Fmt.PNG, Img32.vector, Img32.Extra;
  ...
  var
    img: TImage32;
    path: TPathD;
  begin
    img := TImage32.Create(256,256);
    path :=  Star(PointD(128, 128), 50, 100, 7);
    DrawGlow(img, path, frNonZero, clYellow32, 10);
    DrawPolygon(img, path, frNonZero, clAqua32);
    DrawLine(img, path, 2, clNavy32, esPolygon);
    img.SaveToFile('glowing_star.png');
    img.Free;
  end;
