TImage32.Scale

procedure Scale(s: single); overload;

procedure Scale(sx, sy: single); overload;


Resizes the image relative to its current size.

Example

  uses Img32, Img32.Fmt.PNG;
  ...
  var
    img: TImage32;
    rec: TRect;
  begin
    img := TImage32.Create;
    img.LoadFromFile('fruit.png');
    img.Scale(0.4);
    img.SaveToFile('fruit_small.png');
    img.Free;
  end;
