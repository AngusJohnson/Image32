TLayer32.OuterBounds

property OuterBounds: TRectD; //read only

  Result.Left := fLeft - fOuterMargin;
  Result.Top  :=  fTop - fOuterMargin;
  Result.Right := fLeft + fWidth + fOuterMargin;
  Result.Bottom := fTop + fHeight + fOuterMargin;
